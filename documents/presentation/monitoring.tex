\section{The Monitoring}

\begin{frame}[fragile]
  \frametitle{Pull Image}
  We pull the image for \emph{Prometheus} from the public Docker Hub registry.
  \begin{bashcode}
    $ oc new-app prom/prometheus
    $ oc expose service prometheus
  \end{bashcode}
\end{frame}

\end{frame}
