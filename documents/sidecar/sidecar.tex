\subsection{Status On Apache}

\emph{status.conf}
\begin{minted}{apache}
  <Location /server-status>
  SetHandler server-status
  Order deny,allow
  Allow from all
  </Location> ExtendedStatus On
\end{minted}

and  \emph{Dockerfile}
\begin{minted}{docker}
  RUN a2enmod status
  COPY status.conf /etc/apache2/mods-enabled/
\end{minted}
